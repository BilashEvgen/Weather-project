    def _resize_main_window(self, main_window, width, height):
        screen = main_window.screen()
        if screen is not None:
            geometry = screen.availableGeometry()
            center_x = geometry.x() + (geometry.width() - width) // 2
            center_y = geometry.y() + (geometry.height() - height) // 2
        else:
            center_x = 0
            center_y = 0

        main_window.setGeometry(center_x, center_y, width, height)
        main_window.setFixedSize(width, height)

    def _position_modal_over_weather(self, modal_window, weather_container):
        if modal_window is None or weather_container is None:
            return

        parent = modal_window.parent()
        if not isinstance(parent, widgets.QWidget):
            return

        weather_pos = weather_container.mapTo(parent, core.QPoint(0, 0))
        modal_window.setGeometry(
            weather_pos.x(),
            weather_pos.y(),
            weather_container.width(),
            weather_container.height()
        )
        modal_window.raise_()

    def _scale_widget_tree(self, widget, scale_x, scale_y, excluded_widget=None):
        for child in widget.children():
            if not isinstance(child, widgets.QWidget):
                continue
            if excluded_widget is not None and child is excluded_widget:
                continue
            self._scale_widget(child, scale_x, scale_y)
            self._scale_widget_tree(child, scale_x, scale_y, excluded_widget=excluded_widget)

    def _scale_modal_window(self, modal_window, scale):
        if modal_window is None:
            return

        top_margin = 20
        width = modal_window.width()
        height = modal_window.height()
        new_width = max(1, round(width * scale))
        new_height = max(1, round(height * scale))

        modal_window.setFixedSize(new_width, new_height)
        modal_window.setGeometry(modal_window.x(), modal_window.y(), new_width, new_height)

        header = modal_window.findChild(widgets.QFrame)
        if header is not None:
            header.setFixedWidth(new_width - top_margin)
            header_layout = header.layout()
            if header_layout is not None:
                header_layout.setContentsMargins(0,0,0,0)

    def _scale_widget(self, widget, scale_x, scale_y):
        fixed_width = widget.minimumWidth() == widget.maximumWidth()
        fixed_height = widget.minimumHeight() == widget.maximumHeight()

        width = widget.width()
        height = widget.height()
        if width > 0 and height > 0:
            new_width = max(1, round(width * scale_x))
            new_height = max(1, round(height * scale_y))

            if fixed_width and fixed_height:
                widget.setFixedSize(new_width, new_height)
            elif fixed_width:
                widget.setFixedWidth(new_width)
            elif fixed_height:
                widget.setFixedHeight(new_height)
            else:
                widget.resize(new_width, new_height)

        scale = (scale_x + scale_y) / 2
        self._scale_widget_font(widget, scale)
        self._scale_widget_stylesheet(widget, scale)
        self._scale_widget_pixmap(widget)
        self._scale_widget_icon(widget, scale)

    def _scale_widget_font(self, widget, scale):
        font = widget.font()
        if font is not None and font.pointSize() > 0:
            new_point_size = max(1, round(font.pointSize() * scale))
            if new_point_size != font.pointSize():
                font.setPointSize(new_point_size)
                widget.setFont(font)

    def _scale_widget_stylesheet(self, widget, scale):
        stylesheet = widget.styleSheet()
        if not stylesheet:
            return

        def replace_size(match):
            value = float(match.group(1))
            unit = match.group(2)
            scaled = max(1, round(value * scale))
            return f"font-size: {scaled}{unit}"

        new_stylesheet = re.sub(r"font-size:\s*(\d+\.?\d*)(px|pt)", replace_size, stylesheet)
        if new_stylesheet != stylesheet:
            widget.setStyleSheet(new_stylesheet)

    def _scale_widget_pixmap(self, widget):
        if isinstance(widget, widgets.QLabel) and widget.pixmap() is not None:
            pixmap = widget.pixmap()
            if pixmap is not None and not pixmap.isNull():
                target_size = widget.size()
                scaled_pixmap = pixmap.scaled(target_size, core.Qt.AspectRatioMode.KeepAspectRatio, core.Qt.TransformationMode.SmoothTransformation)
                widget.setPixmap(scaled_pixmap)

    def _scale_widget_icon(self, widget, scale):
        if isinstance(widget, widgets.QAbstractButton):
            icon_size = widget.iconSize()
            if icon_size.isValid() and icon_size.width() > 0 and icon_size.height() > 0:
                new_icon_width = max(1, round(icon_size.width() * scale))
                new_icon_height = max(1, round(icon_size.height() * scale))
                widget.setIconSize(core.QSize(new_icon_width, new_icon_height))
            elif widget.icon().isNull() is False:
                width = widget.width()
                height = widget.height()
                if width > 0 and height > 0:
                    widget.setIconSize(core.QSize(max(1, round(width * scale * 0.4)), max(1, round(height * scale * 0.4))))
